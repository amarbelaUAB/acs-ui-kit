Getting Started

Each component includes a live preview and the corresponding HTML and CSS code.

  1. Copy the HTML snippet.
  2. Paste it into your page.
  3. Copy any required CSS.
  4. Customize as needed.

Installation

Add Bootstrap 5.3 or later, the UAB font stylesheet, and the UAB theme stylesheet to your project.

Requirements
  • Bootstrap 5.3 or later
  • Bootstrap Icons
  • UAB Adobe Fonts stylesheet
  • UAB custom theme stylesheet

JavaScript 
document.addEventListener("DOMContentLoaded", function () {
    const fileInput = document.getElementById("fileAttachment");
    const dropZone = document.getElementById("fileAttachmentDropZone");
    const fileSelection = document.getElementById("fileAttachmentSelection");
    const fileName = document.getElementById("fileAttachmentName");
    const fileSize = document.getElementById("fileAttachmentSize");
    const fileError = document.getElementById("fileAttachmentError");
    const removeButton = document.getElementById("removeFileAttachment");

    const maximumFileSize = 10 * 1024 * 1024;
    const allowedExtensions = ["pdf", "doc", "docx", "png", "jpg", "jpeg"];

    if (!fileInput || !dropZone) {
        return;
    }

    fileInput.addEventListener("change", function () {
        const selectedFile = fileInput.files[0];

        if (selectedFile) {
            handleFile(selectedFile, false);
        }
    });

    dropZone.addEventListener("dragover", function (event) {
        event.preventDefault();
        dropZone.classList.add("border-primary");
    });

    dropZone.addEventListener("dragleave", function () {
        dropZone.classList.remove("border-primary");
    });

    dropZone.addEventListener("drop", function (event) {
        event.preventDefault();
        dropZone.classList.remove("border-primary");

        const droppedFile = event.dataTransfer.files[0];

        if (droppedFile) {
            handleFile(droppedFile, true);
        }
    });

    removeButton.addEventListener("click", function () {
        resetFile();
        hideError();
    });

    function handleFile(file, updateInput) {
        const validationMessage = validateFile(file);

        if (validationMessage) {
            resetFile();
            showError(validationMessage);
            return;
        }

        if (updateInput) {
            const dataTransfer = new DataTransfer();
            dataTransfer.items.add(file);
            fileInput.files = dataTransfer.files;
        }

        hideError();

        fileName.textContent = file.name;
        fileSize.textContent = formatFileSize(file.size);

        dropZone.classList.add("d-none");
        fileSelection.classList.remove("d-none");
    }

    function validateFile(file) {
        const extension = file.name
            .split(".")
            .pop()
            .toLowerCase();

        if (!allowedExtensions.includes(extension)) {
            return "Please select a PDF, DOC, DOCX, PNG, JPG, or JPEG file.";
        }

        if (file.size > maximumFileSize) {
            return "The selected file must be 10 MB or smaller.";
        }

        return "";
    }

    function resetFile() {
        fileInput.value = "";
        fileName.textContent = "";
        fileSize.textContent = "";

        fileSelection.classList.add("d-none");
        dropZone.classList.remove("d-none");
    }

    function showError(message) {
        fileError.textContent = message;
        fileError.classList.remove("d-none");
    }

    function hideError() {
        fileError.textContent = "";
        fileError.classList.add("d-none");
    }

    function formatFileSize(bytes) {
        if (bytes < 1024) {
            return `${bytes} B`;
        }

        if (bytes < 1024 * 1024) {
            return `${(bytes / 1024).toFixed(1)} KB`;
        }

        return `${(bytes / (1024 * 1024)).toFixed(1)} MB`;
    }
});
